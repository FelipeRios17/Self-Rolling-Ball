42void set_duty(mot_struct_t mot_objs) {
43 if(mot_objs.duty < 0) {
44 __HAL_TIM_SET_COMPARE(mot_objs.htim,mot_objs.chan_a,0);
45 __HAL_TIM_SET_COMPARE(mot_objs.htim,mot_objs.chan_b,-mot_objs.duty*mot_objs.ar/100);
46 HAL_Delay(10);
47 }
48 /* duty[%]*AR[ticks]/100[%] represents a number between 0 and AR ticks.
49 * This is based on using a frequency of 20kHz to spin the motor with a
50 * specific prescaler. If duty comes in as a percentage, divide by 100
51 * and multiply by AR to get range of values in ticks.
52 */
53 else {
54 __HAL_TIM_SET_COMPARE(mot_objs.htim,mot_objs.chan_a,mot_objs.duty*mot_objs.ar/100);
55 __HAL_TIM_SET_COMPARE(mot_objs.htim,mot_objs.chan_b,0);
56 HAL_Delay(10);
57 }
