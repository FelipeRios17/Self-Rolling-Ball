101void AsyncWebServerRequest::_onData(void *buf, size_t len){
102 size_t i = 0;
103 while (true) {
104
105 if(_parseState < PARSE_REQ_BODY){
106 // Find new line in buf
107 char *str = (char*)buf;
108 for (i = 0; i < len; i++) {
109 if (str[i] == '\n') {
110 break;
111 }
112 }
113 if (i == len) { // No new line, just add the buffer in _temp
114 char ch = str[len-1];
115 str[len-1] = 0;
116 _temp.reserve(_temp.length()+len);
117 _temp.concat(str);
118 _temp.concat(ch);
119 } else { // Found new line - extract it and parse
120 str[i] = 0; // Terminate the string at the end of the line.
121 _temp.concat(str);
122 _temp.trim();
123 _parseLine();
124 if (++i < len) {
125 // Still have more buffer to process
126 buf = str+i;
127 len-= i;
128 continue;
129 }
130 }
131 } else if(_parseState == PARSE_REQ_BODY){
132 // A handler should be already attached at this point in _parseLine function.
133 // If handler does nothing (_onRequest is NULL), we don't need to really parse the body.
134 const bool needParse = _handler && !_handler->isRequestHandlerTrivial();
135 if(_isMultipart){
136 if(needParse){
137 size_t i;
138 for(i=0; i<len; i++){
139 _parseMultipartPostByte(((uint8_t*)buf)[i], i == len - 1);
140 _parsedLength++;
141 }
142 } else
143 _parsedLength += len;
144 } else {
145 if(_parsedLength == 0){
146 if(_contentType.startsWith("application/x-www-form-urlencoded")){
147 _isPlainPost = true;
148 } else if(_contentType == "text/plain" && __is_param_char(((char*)buf)[0])){
149 size_t i = 0;
150 while (i<len && __is_param_char(((char*)buf)[i++]));
151 if(i < len && ((char*)buf)[i-1] == '='){
152 _isPlainPost = true;
153 }
154 }
155 }
156 if(!_isPlainPost) {
157 //check if authenticated before calling the body
158 if(_handler) _handler->handleBody(this, (uint8_t*)buf, len, _parsedLength, _contentLength);
159 _parsedLength += len;
160 } else if(needParse) {
161 size_t i;
162 for(i=0; i<len; i++){
163 _parsedLength++;
164 _parsePlainPostChar(((uint8_t*)buf)[i]);
165 }
166 } else {
167 _parsedLength += len;
168 }
169 }
170 if(_parsedLength == _contentLength){
171 _parseState = PARSE_REQ_END;
172 //check if authenticated before calling handleRequest and request auth instead
173 if(_handler) _handler->handleRequest(this);
174 else send(501);
175 }
176 }
177 break;
178 }
179}

240void AsyncWebServerRequest::_addGetParams(const String& params){
241 size_t start = 0;
242 while (start < params.length()){
243 int end = params.indexOf('&', start);
244 if (end < 0) end = params.length();
245 int equal = params.indexOf('=', start);
246 if (equal < 0 || equal > end) equal = end;
247 String name = params.substring(start, equal);
248 String value = equal + 1 < end ? params.substring(equal + 1, end) : String();
249 _addParam(new AsyncWebParameter(urlDecode(name), urlDecode(value)));
250 start = end + 1;
251 }

254bool AsyncWebServerRequest::_parseReqHead(){
255 // Split the head into method, url and version
256 int index = _temp.indexOf(' ');
257 String m = _temp.substring(0, index);
258 index = _temp.indexOf(' ', index+1);
259 String u = _temp.substring(m.length()+1, index);
260 _temp = _temp.substring(index+1);
261
262 if(m == "GET"){
263 _method = HTTP_GET;
264 } else if(m == "POST"){
265 _method = HTTP_POST;
266 } else if(m == "DELETE"){
267 _method = HTTP_DELETE;
268 } else if(m == "PUT"){
269 _method = HTTP_PUT;
270 } else if(m == "PATCH"){
271 _method = HTTP_PATCH;
272 } else if(m == "HEAD"){
273 _method = HTTP_HEAD;
274 } else if(m == "OPTIONS"){
275 _method = HTTP_OPTIONS;
276 }
277
278 String g = String();
279 index = u.indexOf('?');
280 if(index > 0){
281 g = u.substring(index +1);
282 u = u.substring(0, index);
283 }
284 _url = urlDecode(u);
285 _addGetParams(g);
286
287 if(!_temp.startsWith("HTTP/1.0"))
288 _version = 1;
289
290 _temp = String();
291 return true;
292}

294bool strContains(String src, String find, bool mindcase = true) {
295 int pos=0, i=0;
296 const int slen = src.length();
297 const int flen = find.length();
298
299 if (slen < flen) return false;
300 while (pos <= (slen - flen)) {
301 for (i=0; i < flen; i++) {
302 if (mindcase) {
303 if (src[pos+i] != find[i]) i = flen + 1; // no match
304 } else if (tolower(src[pos+i]) != tolower(find[i])) i = flen + 1; // no match
305 }
306 if (i == flen) return true;
307 pos++;
308 }
309 return false;
310}

312bool AsyncWebServerRequest::_parseReqHeader(){
313 int index = _temp.indexOf(':');
314 if(index){
315 String name = _temp.substring(0, index);
316 String value = _temp.substring(index + 2);
317 if(name.equalsIgnoreCase("Host")){
318 _host = value;
319 } else if(name.equalsIgnoreCase("Content-Type")){
320 _contentType = value.substring(0, value.indexOf(';'));
321 if (value.startsWith("multipart/")){
322 _boundary = value.substring(value.indexOf('=')+1);
323 _boundary.replace("\"","");
324 _isMultipart = true;
325 }
326 } else if(name.equalsIgnoreCase("Content-Length")){
327 _contentLength = atoi(value.c_str());
328 } else if(name.equalsIgnoreCase("Expect") && value == "100-continue"){
329 _expectingContinue = true;
330 } else if(name.equalsIgnoreCase("Authorization")){
331 if(value.length() > 5 && value.substring(0,5).equalsIgnoreCase("Basic")){
332 _authorization = value.substring(6);
333 } else if(value.length() > 6 && value.substring(0,6).equalsIgnoreCase("Digest")){
334 _isDigest = true;
335 _authorization = value.substring(7);
336 }
337 } else {
338 if(name.equalsIgnoreCase("Upgrade") && value.equalsIgnoreCase("websocket")){
339 // WebSocket request can be uniquely identified by header: [Upgrade: websocket]
340 _reqconntype = RCT_WS;
341 } else {
342 if(name.equalsIgnoreCase("Accept") && strContains(value, "text/event-stream", false)){
343 // WebEvent request can be uniquely identified by header: [Accept: text/event-stream]
344 _reqconntype = RCT_EVENT;
345 }
346 }
347 }
348 _headers.add(new AsyncWebHeader(name, value));
349 }
350 _temp = String();
351 return true;
352}

381 EXPECT_BOUNDARY,
382 PARSE_HEADERS,
383 WAIT_FOR_RETURN1,
384 EXPECT_FEED1,
385 EXPECT_DASH1,
386 EXPECT_DASH2,
387 BOUNDARY_OR_DATA,
388 DASH3_OR_RETURN2,

394void AsyncWebServerRequest::_parseMultipartPostByte(uint8_t data, bool last){
395#define itemWriteByte(b) do { _itemSize++; if(_itemIsFile) _handleUploadByte(b, last); else _itemValue+=(char)(b); } while(0)
396
397 if(!_parsedLength){
398 _multiParseState = EXPECT_BOUNDARY;
399 _temp = String();
400 _itemName = String();
401 _itemFilename = String();
402 _itemType = String();
403 }
404
405 if(_multiParseState == WAIT_FOR_RETURN1){
406 if(data != '\r'){
407 itemWriteByte(data);
408 } else {
409 _multiParseState = EXPECT_FEED1;
410 }
411 } else if(_multiParseState == EXPECT_BOUNDARY){
412 if(_parsedLength < 2 && data != '-'){
413 _multiParseState = PARSE_ERROR;
414 return;
415 } else if(_parsedLength - 2 < _boundary.length() && _boundary.c_str()[_parsedLength - 2] != data){
416 _multiParseState = PARSE_ERROR;
417 return;
418 } else if(_parsedLength - 2 == _boundary.length() && data != '\r'){
419 _multiParseState = PARSE_ERROR;
420 return;
421 } else if(_parsedLength - 3 == _boundary.length()){
422 if(data != '\n'){
423 _multiParseState = PARSE_ERROR;
424 return;
425 }
426 _multiParseState = PARSE_HEADERS;
427 _itemIsFile = false;
428 }
429 } else if(_multiParseState == PARSE_HEADERS){
430 if((char)data != '\r' && (char)data != '\n')
431 _temp += (char)data;
432 if((char)data == '\n'){
433 if(_temp.length()){
434 if(_temp.length() > 12 && _temp.substring(0, 12).equalsIgnoreCase("Content-Type")){
435 _itemType = _temp.substring(14);
436 _itemIsFile = true;
437 } else if(_temp.length() > 19 && _temp.substring(0, 19).equalsIgnoreCase("Content-Disposition")){
438 _temp = _temp.substring(_temp.indexOf(';') + 2);
439 while(_temp.indexOf(';') > 0){
440 String name = _temp.substring(0, _temp.indexOf('='));
441 String nameVal = _temp.substring(_temp.indexOf('=') + 2, _temp.indexOf(';') - 1);
442 if(name == "name"){
443 _itemName = nameVal;
444 } else if(name == "filename"){
445 _itemFilename = nameVal;
446 _itemIsFile = true;
447 }
448 _temp = _temp.substring(_temp.indexOf(';') + 2);
449 }
450 String name = _temp.substring(0, _temp.indexOf('='));
451 String nameVal = _temp.substring(_temp.indexOf('=') + 2, _temp.length() - 1);
452 if(name == "name"){
453 _itemName = nameVal;
454 } else if(name == "filename"){
455 _itemFilename = nameVal;
456 _itemIsFile = true;
457 }
458 }
459 _temp = String();
460 } else {
461 _multiParseState = WAIT_FOR_RETURN1;
462 //value starts from here
463 _itemSize = 0;
464 _itemStartIndex = _parsedLength;
465 _itemValue = String();
466 if(_itemIsFile){
467 if(_itemBuffer)
468 free(_itemBuffer);
469 _itemBuffer = (uint8_t*)malloc(1460);
470 if(_itemBuffer == NULL){
471 _multiParseState = PARSE_ERROR;
472 return;
473 }
474 _itemBufferIndex = 0;
475 }
476 }
477 }
478 } else if(_multiParseState == EXPECT_FEED1){
479 if(data != '\n'){
480 _multiParseState = WAIT_FOR_RETURN1;
481 itemWriteByte('\r'); _parseMultipartPostByte(data, last);
482 } else {
483 _multiParseState = EXPECT_DASH1;
484 }
485 } else if(_multiParseState == EXPECT_DASH1){
486 if(data != '-'){
487 _multiParseState = WAIT_FOR_RETURN1;
488 itemWriteByte('\r'); itemWriteByte('\n'); _parseMultipartPostByte(data, last);
489 } else {
490 _multiParseState = EXPECT_DASH2;
491 }
492 } else if(_multiParseState == EXPECT_DASH2){
493 if(data != '-'){
494 _multiParseState = WAIT_FOR_RETURN1;
495 itemWriteByte('\r'); itemWriteByte('\n'); itemWriteByte('-'); _parseMultipartPostByte(data, last);
496 } else {
497 _multiParseState = BOUNDARY_OR_DATA;
498 _boundaryPosition = 0;
499 }
500 } else if(_multiParseState == BOUNDARY_OR_DATA){
501 if(_boundaryPosition < _boundary.length() && _boundary.c_str()[_boundaryPosition] != data){
502 _multiParseState = WAIT_FOR_RETURN1;
503 itemWriteByte('\r'); itemWriteByte('\n'); itemWriteByte('-'); itemWriteByte('-');
504 uint8_t i;
505 for(i=0; i<_boundaryPosition; i++)
506 itemWriteByte(_boundary.c_str()[i]);
507 _parseMultipartPostByte(data, last);
508 } else if(_boundaryPosition == _boundary.length() - 1){
509 _multiParseState = DASH3_OR_RETURN2;
510 if(!_itemIsFile){
511 _addParam(new AsyncWebParameter(_itemName, _itemValue, true));
512 } else {
513 if(_itemSize){
514 //check if authenticated before calling the upload
515 if(_handler) _handler->handleUpload(this, _itemFilename, _itemSize - _itemBufferIndex, _itemBuffer, _itemBufferIndex, true);
516 _itemBufferIndex = 0;
517 _addParam(new AsyncWebParameter(_itemName, _itemFilename, true, true, _itemSize));
518 }
519 free(_itemBuffer);
520 _itemBuffer = NULL;
521 }
522
523 } else {
524 _boundaryPosition++;
525 }
526 } else if(_multiParseState == DASH3_OR_RETURN2){
527 if(data == '-' && (_contentLength - _parsedLength - 4) != 0){
528 //os_printf("ERROR: The parser got to the end of the POST but is expecting %u bytes more!\nDrop an issue so we can have more info on the matter!\n", _contentLength - _parsedLength - 4);
529 _contentLength = _parsedLength + 4;//lets close the request gracefully
530 }
531 if(data == '\r'){
532 _multiParseState = EXPECT_FEED2;
533 } else if(data == '-' && _contentLength == (_parsedLength + 4)){
534 _multiParseState = PARSING_FINISHED;
535 } else {
536 _multiParseState = WAIT_FOR_RETURN1;
537 itemWriteByte('\r'); itemWriteByte('\n'); itemWriteByte('-'); itemWriteByte('-');
538 uint8_t i; for(i=0; i<_boundary.length(); i++) itemWriteByte(_boundary.c_str()[i]);
539 _parseMultipartPostByte(data, last);
540 }
541 } else if(_multiParseState == EXPECT_FEED2){
542 if(data == '\n'){
543 _multiParseState = PARSE_HEADERS;
544 _itemIsFile = false;
545 } else {
546 _multiParseState = WAIT_FOR_RETURN1;
547 itemWriteByte('\r'); itemWriteByte('\n'); itemWriteByte('-'); itemWriteByte('-');
548 uint8_t i; for(i=0; i<_boundary.length(); i++) itemWriteByte(_boundary.c_str()[i]);
549 itemWriteByte('\r'); _parseMultipartPostByte(data, last);
550 }
551 }
552}

554void AsyncWebServerRequest::_parseLine(){
555 if(_parseState == PARSE_REQ_START){
556 if(!_temp.length()){
557 _parseState = PARSE_REQ_FAIL;
558 _client->close();
559 } else {
560 _parseReqHead();
561 _parseState = PARSE_REQ_HEADERS;
562 }
563 return;
564 }
565
566 if(_parseState == PARSE_REQ_HEADERS){
567 if(!_temp.length()){
568 //end of headers
569 _server->_rewriteRequest(this);
570 _server->_attachHandler(this);
571 _removeNotInterestingHeaders();
572 if(_expectingContinue){
573 const char * response = "HTTP/1.1 100 Continue\r\n\r\n";
574 _client->write(response, os_strlen(response));
575 }
576 //check handler for authentication
577 if(_contentLength){
578 _parseState = PARSE_REQ_BODY;
579 } else {
580 _parseState = PARSE_REQ_END;
581 if(_handler) _handler->handleRequest(this);
582 else send(501);
583 }
584 } else _parseReqHeader();
585 }

825bool AsyncWebServerRequest::authenticate(const char * hash){
826 if(!_authorization.length() || hash == NULL)
827 return false;
828
829 if(_isDigest){
830 String hStr = String(hash);
831 int separator = hStr.indexOf(":");
832 if(separator <= 0)
833 return false;
834 String username = hStr.substring(0, separator);
835 hStr = hStr.substring(separator + 1);
836 separator = hStr.indexOf(":");
837 if(separator <= 0)
838 return false;
839 String realm = hStr.substring(0, separator);
840 hStr = hStr.substring(separator + 1);
841 return checkDigestAuthentication(_authorization.c_str(), methodToString(), username.c_str(), hStr.c_str(), realm.c_str(), true, NULL, NULL, NULL);
842 }
843
844 return (_authorization.equals(hash));

955String AsyncWebServerRequest::urlDecode(const String& text) const {
956 char temp[] = "0x00";
957 unsigned int len = text.length();
958 unsigned int i = 0;
959 String decoded = String();
960 decoded.reserve(len); // Allocate the string internal buffer - never longer from source text
961 while (i < len){
962 char decodedChar;
963 char encodedChar = text.charAt(i++);
964 if ((encodedChar == '%') && (i + 1 < len)){
965 temp[2] = text.charAt(i++);
966 temp[3] = text.charAt(i++);
967 decodedChar = strtol(temp, NULL, 16);
968 } else if (encodedChar == '+') {
969 decodedChar = ' ';
970 } else {
971 decodedChar = encodedChar; // normal ascii char
972 }
973 decoded.concat(decodedChar);
974 }
975 return decoded;
