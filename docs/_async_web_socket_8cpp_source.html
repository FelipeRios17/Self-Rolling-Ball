43size_t webSocketSendFrame(AsyncClient *client, bool final, uint8_t opcode, bool mask, uint8_t *data, size_t len){
44 if(!client->canSend())
45 return 0;
46 size_t space = client->space();
47 if(space < 2)
48 return 0;
49 uint8_t mbuf[4] = {0,0,0,0};
50 uint8_t headLen = 2;
51 if(len && mask){
52 headLen += 4;
53 mbuf[0] = rand() % 0xFF;
54 mbuf[1] = rand() % 0xFF;
55 mbuf[2] = rand() % 0xFF;
56 mbuf[3] = rand() % 0xFF;
57 }
58 if(len > 125)
59 headLen += 2;
60 if(space < headLen)
61 return 0;
62 space -= headLen;
63
64 if(len > space) len = space;
65
66 uint8_t *buf = (uint8_t*)malloc(headLen);
67 if(buf == NULL){
68 //os_printf("could not malloc %u bytes for frame header\n", headLen);
69 return 0;
70 }
71
72 buf[0] = opcode & 0x0F;
73 if(final)
74 buf[0] |= 0x80;
75 if(len < 126)
76 buf[1] = len & 0x7F;
77 else {
78 buf[1] = 126;
79 buf[2] = (uint8_t)((len >> 8) & 0xFF);
80 buf[3] = (uint8_t)(len & 0xFF);
81 }
82 if(len && mask){
83 buf[1] |= 0x80;
84 memcpy(buf + (headLen - 4), mbuf, 4);
85 }
86 if(client->add((const char *)buf, headLen) != headLen){
87 //os_printf("error adding %lu header bytes\n", headLen);
88 free(buf);
89 return 0;
90 }
91 free(buf);
92
93 if(len){
94 if(len && mask){
95 size_t i;
96 for(i=0;i<len;i++)
97 data[i] = data[i] ^ mbuf[i%4];
98 }
99 if(client->add((const char *)data, len) != len){
100 //os_printf("error adding %lu data bytes\n", len);
101 return 0;
102 }
103 }
104 if(!client->send()){
105 //os_printf("error sending frame: %lu\n", headLen+len);
106 return 0;
107 }
108 return len;
109}

233class AsyncWebSocketControl {
234 private:
235 uint8_t _opcode;
236 uint8_t *_data;
237 size_t _len;
238 bool _mask;
239 bool _finished;
240 public:
241 AsyncWebSocketControl(uint8_t opcode, uint8_t *data=NULL, size_t len=0, bool mask=false)
242 :_opcode(opcode)
243 ,_len(len)
244 ,_mask(len && mask)
245 ,_finished(false)
246 {
247 if(data == NULL)
248 _len = 0;
249 if(_len){
250 if(_len > 125)
251 _len = 125;
252 _data = (uint8_t*)malloc(_len);
253 if(_data == NULL)
254 _len = 0;
255 else memcpy(_data, data, len);
256 } else _data = NULL;
257 }
258 virtual ~AsyncWebSocketControl(){
259 if(_data != NULL)
260 free(_data);
261 }
262 virtual bool finished() const { return _finished; }
263 uint8_t opcode(){ return _opcode; }
264 uint8_t len(){ return _len + 2; }
265 size_t send(AsyncClient *client){
266 _finished = true;
267 return webSocketSendFrame(client, true, _opcode & 0x0F, _mask, _data, _len);
268 }
269};

319 size_t AsyncWebSocketBasicMessage::send(AsyncClient *client) {
320 if(_status != WS_MSG_SENDING)
321 return 0;
322 if(_acked < _ack){
323 return 0;
324 }
325 if(_sent == _len){
326 if(_acked == _ack)
327 _status = WS_MSG_SENT;
328 return 0;
329 }
330 if(_sent > _len){
331 _status = WS_MSG_ERROR;
332 return 0;
333 }
334
335 size_t toSend = _len - _sent;
336 size_t window = webSocketSendFrameWindow(client);
337
338 if(window < toSend) {
339 toSend = window;
340 }
341
342 _sent += toSend;
343 _ack += toSend + ((toSend < 126)?2:4) + (_mask * 4);
344
345 bool final = (_sent == _len);
346 uint8_t* dPtr = (uint8_t*)(_data + (_sent - toSend));
347 uint8_t opCode = (toSend && _sent == toSend)?_opcode:(uint8_t)WS_CONTINUATION;
348
349 size_t sent = webSocketSendFrame(client, final, opCode, _mask, dPtr, toSend);
350 _status = WS_MSG_SENDING;
351 if(toSend && sent != toSend){
352 _sent -= (toSend - sent);
353 _ack -= (toSend - sent);
354 }
355 return sent;
356}

466AsyncWebSocketClient::AsyncWebSocketClient(AsyncWebServerRequest *request, AsyncWebSocket *server)
467 : _controlQueue(LinkedList<AsyncWebSocketControl *>([](AsyncWebSocketControl *c){ delete c; }))
468 , _messageQueue(LinkedList<AsyncWebSocketMessage *>([](AsyncWebSocketMessage *m){ delete m; }))
469 , _tempObject(NULL)
470{
471 _client = request->client();
472 _server = server;
473 _clientId = _server->_getNextId();
474 _status = WS_CONNECTED;
475 _pstate = 0;
476 _lastMessageTime = millis();
477 _keepAlivePeriod = 0;
478 _client->setRxTimeout(0);
479 _client->onError([](void *r, AsyncClient* c, int8_t error){ (void)c; ((AsyncWebSocketClient*)(r))->_onError(error); }, this);
480 _client->onAck([](void *r, AsyncClient* c, size_t len, uint32_t time){ (void)c; ((AsyncWebSocketClient*)(r))->_onAck(len, time); }, this);
481 _client->onDisconnect([](void *r, AsyncClient* c){ ((AsyncWebSocketClient*)(r))->_onDisconnect(); delete c; }, this);
482 _client->onTimeout([](void *r, AsyncClient* c, uint32_t time){ (void)c; ((AsyncWebSocketClient*)(r))->_onTimeout(time); }, this);
483 _client->onData([](void *r, AsyncClient* c, void *buf, size_t len){ (void)c; ((AsyncWebSocketClient*)(r))->_onData(buf, len); }, this);
484 _client->onPoll([](void *r, AsyncClient* c){ (void)c; ((AsyncWebSocketClient*)(r))->_onPoll(); }, this);
485 _server->_addClient(this);
486 _server->_handleEvent(this, WS_EVT_CONNECT, request, NULL, 0);
487 delete request;
488}

496void AsyncWebSocketClient::_onAck(size_t len, uint32_t time){
497 _lastMessageTime = millis();
498 if(!_controlQueue.isEmpty()){
499 auto head = _controlQueue.front();
500 if(head->finished()){
501 len -= head->len();
502 if(_status == WS_DISCONNECTING && head->opcode() == WS_DISCONNECT){
503 _controlQueue.remove(head);
504 _status = WS_DISCONNECTED;
505 _client->close(true);
506 return;
507 }
508 _controlQueue.remove(head);
509 }
510 }
511 if(len && !_messageQueue.isEmpty()){
512 _messageQueue.front()->ack(len, time);
513 }
514 _server->_cleanBuffers();
515 _runQueue();
516}
517
518void AsyncWebSocketClient::_onPoll(){
519 if(_client->canSend() && (!_controlQueue.isEmpty() || !_messageQueue.isEmpty())){
520 _runQueue();
521 } else if(_keepAlivePeriod > 0 && _controlQueue.isEmpty() && _messageQueue.isEmpty() && (millis() - _lastMessageTime) >= _keepAlivePeriod){
522 ping((uint8_t *)AWSC_PING_PAYLOAD, AWSC_PING_PAYLOAD_LEN);
523 }
524}
525
526void AsyncWebSocketClient::_runQueue(){
527 while(!_messageQueue.isEmpty() && _messageQueue.front()->finished()){
528 _messageQueue.remove(_messageQueue.front());
529 }
530
531 if(!_controlQueue.isEmpty() && (_messageQueue.isEmpty() || _messageQueue.front()->betweenFrames()) && webSocketSendFrameWindow(_client) > (size_t)(_controlQueue.front()->len() - 1)){
532 _controlQueue.front()->send(_client);
533 } else if(!_messageQueue.isEmpty() && _messageQueue.front()->betweenFrames() && webSocketSendFrameWindow(_client)){
534 _messageQueue.front()->send(_client);
535 }
536}

568void AsyncWebSocketClient::close(uint16_t code, const char * message){
569 if(_status != WS_CONNECTED)
570 return;
571 if(code){
572 uint8_t packetLen = 2;
573 if(message != NULL){
574 size_t mlen = strlen(message);
575 if(mlen > 123) mlen = 123;
576 packetLen += mlen;
577 }
578 char * buf = (char*)malloc(packetLen);
579 if(buf != NULL){
580 buf[0] = (uint8_t)(code >> 8);
581 buf[1] = (uint8_t)(code & 0xFF);
582 if(message != NULL){
583 memcpy(buf+2, message, packetLen -2);
584 }
585 _queueControl(new AsyncWebSocketControl(WS_DISCONNECT,(uint8_t*)buf,packetLen));
586 free(buf);
587 return;
588 }
589 }
590 _queueControl(new AsyncWebSocketControl(WS_DISCONNECT));
591}

610void AsyncWebSocketClient::_onData(void *pbuf, size_t plen){
611 _lastMessageTime = millis();
612 uint8_t *data = (uint8_t*)pbuf;
613 while(plen > 0){
614 if(!_pstate){
615 const uint8_t *fdata = data;
616 _pinfo.index = 0;
617 _pinfo.final = (fdata[0] & 0x80) != 0;
618 _pinfo.opcode = fdata[0] & 0x0F;
619 _pinfo.masked = (fdata[1] & 0x80) != 0;
620 _pinfo.len = fdata[1] & 0x7F;
621 data += 2;
622 plen -= 2;
623 if(_pinfo.len == 126){
624 _pinfo.len = fdata[3] | (uint16_t)(fdata[2]) << 8;
625 data += 2;
626 plen -= 2;
627 } else if(_pinfo.len == 127){
628 _pinfo.len = fdata[9] | (uint16_t)(fdata[8]) << 8 | (uint32_t)(fdata[7]) << 16 | (uint32_t)(fdata[6]) << 24 | (uint64_t)(fdata[5]) << 32 | (uint64_t)(fdata[4]) << 40 | (uint64_t)(fdata[3]) << 48 | (uint64_t)(fdata[2]) << 56;
629 data += 8;
630 plen -= 8;
631 }
632
633 if(_pinfo.masked){
634 memcpy(_pinfo.mask, data, 4);
635 data += 4;
636 plen -= 4;
637 }
638 }
639
640 const size_t datalen = std::min((size_t)(_pinfo.len - _pinfo.index), plen);
641 const auto datalast = data[datalen];
642
643 if(_pinfo.masked){
644 for(size_t i=0;i<datalen;i++)
645 data[i] ^= _pinfo.mask[(_pinfo.index+i)%4];
646 }
647
648 if((datalen + _pinfo.index) < _pinfo.len){
649 _pstate = 1;
650
651 if(_pinfo.index == 0){
652 if(_pinfo.opcode){
653 _pinfo.message_opcode = _pinfo.opcode;
654 _pinfo.num = 0;
655 } else _pinfo.num += 1;
656 }
657 _server->_handleEvent(this, WS_EVT_DATA, (void *)&_pinfo, (uint8_t*)data, datalen);
658
659 _pinfo.index += datalen;
660 } else if((datalen + _pinfo.index) == _pinfo.len){
661 _pstate = 0;
662 if(_pinfo.opcode == WS_DISCONNECT){
663 if(datalen){
664 uint16_t reasonCode = (uint16_t)(data[0] << 8) + data[1];
665 char * reasonString = (char*)(data+2);
666 if(reasonCode > 1001){
667 _server->_handleEvent(this, WS_EVT_ERROR, (void *)&reasonCode, (uint8_t*)reasonString, strlen(reasonString));
668 }
669 }
670 if(_status == WS_DISCONNECTING){
671 _status = WS_DISCONNECTED;
672 _client->close(true);
673 } else {
674 _status = WS_DISCONNECTING;
675 _client->ackLater();
676 _queueControl(new AsyncWebSocketControl(WS_DISCONNECT, data, datalen));
677 }
678 } else if(_pinfo.opcode == WS_PING){
679 _queueControl(new AsyncWebSocketControl(WS_PONG, data, datalen));
680 } else if(_pinfo.opcode == WS_PONG){
681 if(datalen != AWSC_PING_PAYLOAD_LEN || memcmp(AWSC_PING_PAYLOAD, data, AWSC_PING_PAYLOAD_LEN) != 0)
682 _server->_handleEvent(this, WS_EVT_PONG, NULL, data, datalen);
683 } else if(_pinfo.opcode < 8){//continuation or text/binary frame
684 _server->_handleEvent(this, WS_EVT_DATA, (void *)&_pinfo, data, datalen);
685 }
686 } else {
687 //os_printf("frame error: len: %u, index: %llu, total: %llu\n", datalen, _pinfo.index, _pinfo.len);
688 //what should we do?
689 break;
690 }
691
692 // restore byte as _handleEvent may have added a null terminator i.e., data[len] = 0;
693 if (datalen > 0)
694 data[datalen] = datalast;
695
696 data += datalen;
697 plen -= datalen;
698 }
699}

701size_t AsyncWebSocketClient::printf(const char *format, ...) {
702 va_list arg;
703 va_start(arg, format);
704 char* temp = new char[MAX_PRINTF_LEN];
705 if(!temp){
706 va_end(arg);
707 return 0;
708 }
709 char* buffer = temp;
710 size_t len = vsnprintf(temp, MAX_PRINTF_LEN, format, arg);
711 va_end(arg);
712
713 if (len > (MAX_PRINTF_LEN - 1)) {
714 buffer = new char[len + 1];
715 if (!buffer) {
716 delete[] temp;
717 return 0;
718 }
719 va_start(arg, format);
720 vsnprintf(buffer, len + 1, format, arg);
721 va_end(arg);
722 }
723 text(buffer, len);
724 if (buffer != temp) {
725 delete[] buffer;
726 }
727 delete[] temp;
728 return len;
729}

1237 * Authentication code from https://github.com/Links2004/arduinoWebSockets/blob/master/src/WebSockets.cpp#L480

1240AsyncWebSocketResponse::AsyncWebSocketResponse(const String& key, AsyncWebSocket *server){
1241 _server = server;
1242 _code = 101;
1243 _sendContentLength = false;
1244
1245 uint8_t * hash = (uint8_t*)malloc(20);
1246 if(hash == NULL){
1247 _state = RESPONSE_FAILED;
1248 return;
1249 }
1250 char * buffer = (char *) malloc(33);
1251 if(buffer == NULL){
1252 free(hash);
1253 _state = RESPONSE_FAILED;
1254 return;

1261 mbedtls_sha1_init(&ctx);
1262 mbedtls_sha1_starts_ret(&ctx);
1263 mbedtls_sha1_update_ret(&ctx, (const unsigned char*)key.c_str(), key.length());

1265 mbedtls_sha1_free(&ctx);
1266#endif
1267 base64_encodestate _state;
1268 base64_init_encodestate(&_state);
1269 int len = base64_encode_block((const char *) hash, 20, buffer, &_state);
1270 len = base64_encode_blockend((buffer + len), &_state);

1272 addHeader(WS_STR_UPGRADE, "websocket");
