118static bool _remove_events_with_arg(void * arg){
119 lwip_event_packet_t * first_packet = NULL;
120 lwip_event_packet_t * packet = NULL;
121
122 if(!_async_queue){
123 return false;
124 }
125 //figure out which is the first packet so we can keep the order
126 while(!first_packet){
127 if(xQueueReceive(_async_queue, &first_packet, 0) != pdPASS){
128 return false;
129 }
130 //discard packet if matching
131 if((int)first_packet->arg == (int)arg){
132 free(first_packet);
133 first_packet = NULL;
134 //return first packet to the back of the queue
135 } else if(xQueueSend(_async_queue, &first_packet, portMAX_DELAY) != pdPASS){
136 return false;
137 }
138 }
139
140 while(xQueuePeek(_async_queue, &packet, 0) == pdPASS && packet != first_packet){
141 if(xQueueReceive(_async_queue, &packet, 0) != pdPASS){
142 return false;
143 }
144 if((int)packet->arg == (int)arg){
145 free(packet);
146 packet = NULL;
147 } else if(xQueueSend(_async_queue, &packet, portMAX_DELAY) != pdPASS){
148 return false;
149 }
150 }
151 return true;
152}

154static void _handle_async_event(lwip_event_packet_t * e){
155 if(e->arg == NULL){
156 // do nothing when arg is NULL
157 //ets_printf("event arg == NULL: 0x%08x\n", e->recv.pcb);
158 } else if(e->event == LWIP_TCP_CLEAR){
159 _remove_events_with_arg(e->arg);
160 } else if(e->event == LWIP_TCP_RECV){
161 //ets_printf("-R: 0x%08x\n", e->recv.pcb);
162 AsyncClient::_s_recv(e->arg, e->recv.pcb, e->recv.pb, e->recv.err);
163 } else if(e->event == LWIP_TCP_FIN){
164 //ets_printf("-F: 0x%08x\n", e->fin.pcb);
165 AsyncClient::_s_fin(e->arg, e->fin.pcb, e->fin.err);
166 } else if(e->event == LWIP_TCP_SENT){
167 //ets_printf("-S: 0x%08x\n", e->sent.pcb);
168 AsyncClient::_s_sent(e->arg, e->sent.pcb, e->sent.len);
169 } else if(e->event == LWIP_TCP_POLL){
170 //ets_printf("-P: 0x%08x\n", e->poll.pcb);
171 AsyncClient::_s_poll(e->arg, e->poll.pcb);
172 } else if(e->event == LWIP_TCP_ERROR){
173 //ets_printf("-E: 0x%08x %d\n", e->arg, e->error.err);
174 AsyncClient::_s_error(e->arg, e->error.err);
175 } else if(e->event == LWIP_TCP_CONNECTED){
176 //ets_printf("C: 0x%08x 0x%08x %d\n", e->arg, e->connected.pcb, e->connected.err);
177 AsyncClient::_s_connected(e->arg, e->connected.pcb, e->connected.err);
178 } else if(e->event == LWIP_TCP_ACCEPT){
179 //ets_printf("A: 0x%08x 0x%08x\n", e->arg, e->accept.client);
180 AsyncServer::_s_accepted(e->arg, e->accept.client);
181 } else if(e->event == LWIP_TCP_DNS){
182 //ets_printf("D: 0x%08x %s = %s\n", e->arg, e->dns.name, ipaddr_ntoa(&e->dns.addr));
183 AsyncClient::_s_dns_found(e->dns.name, &e->dns.addr, e->arg);
184 }
185 free((void*)(e));
186}

188static void _async_service_task(void *pvParameters){
189 lwip_event_packet_t * packet = NULL;
190 for (;;) {
191 if(_get_async_event(&packet)){
192#if CONFIG_ASYNC_TCP_USE_WDT
193 if(esp_task_wdt_add(NULL) != ESP_OK){
194 log_e("Failed to add async task to WDT");
195 }
196#endif
197 _handle_async_event(packet);
198#if CONFIG_ASYNC_TCP_USE_WDT
199 if(esp_task_wdt_delete(NULL) != ESP_OK){
200 log_e("Failed to remove loop task from WDT");
201 }
202#endif
203 }
204 }
205 vTaskDelete(NULL);
206 _async_service_task_handle = NULL;
207}

268static int8_t _tcp_recv(void * arg, struct tcp_pcb * pcb, struct pbuf *pb, int8_t err) {
269 lwip_event_packet_t * e = (lwip_event_packet_t *)malloc(sizeof(lwip_event_packet_t));
270 e->arg = arg;
271 if(pb){
272 //ets_printf("+R: 0x%08x\n", pcb);
273 e->event = LWIP_TCP_RECV;
274 e->recv.pcb = pcb;
275 e->recv.pb = pb;
276 e->recv.err = err;
277 } else {
278 //ets_printf("+F: 0x%08x\n", pcb);
279 e->event = LWIP_TCP_FIN;
280 e->fin.pcb = pcb;
281 e->fin.err = err;
282 //close the PCB in LwIP thread
283 AsyncClient::_s_lwip_fin(e->arg, e->fin.pcb, e->fin.err);
284 }
285 if (!_send_async_event(&e)) {
286 free((void*)(e));
287 }

374static err_t _tcp_output_api(struct tcpip_api_call_data *api_call_msg){

543AsyncClient::AsyncClient(tcp_pcb* pcb)
544: _connect_cb(0)
545, _connect_cb_arg(0)
546, _discard_cb(0)
547, _discard_cb_arg(0)
548, _sent_cb(0)
549, _sent_cb_arg(0)
550, _error_cb(0)
551, _error_cb_arg(0)
552, _recv_cb(0)
553, _recv_cb_arg(0)
554, _pb_cb(0)
555, _pb_cb_arg(0)
556, _timeout_cb(0)
557, _timeout_cb_arg(0)
558, _pcb_busy(false)
559, _pcb_sent_at(0)
560, _ack_pcb(true)
561, _rx_last_packet(0)
562, _rx_since_timeout(0)
563, _ack_timeout(ASYNC_MAX_ACK_TIME)
564, _connect_port(0)
565, prev(NULL)
566, next(NULL)
567{
568 _pcb = pcb;
569 _closed_slot = -1;
570 if(_pcb){
571 _allocate_closed_slot();
572 _rx_last_packet = millis();
573 tcp_arg(_pcb, this);
574 tcp_recv(_pcb, &_tcp_recv);
575 tcp_sent(_pcb, &_tcp_sent);
576 tcp_err(_pcb, &_tcp_error);
577 tcp_poll(_pcb, &_tcp_poll, 1);
578 }
579}
580
581AsyncClient::~AsyncClient(){
582 if(_pcb) {
583 _close();
584 }
585 _free_closed_slot();
586}

592AsyncClient& AsyncClient::operator=(const AsyncClient& other){

594 _close();

601 tcp_arg(_pcb, this);
602 tcp_recv(_pcb, &_tcp_recv);
603 tcp_sent(_pcb, &_tcp_sent);
604 tcp_err(_pcb, &_tcp_error);
605 tcp_poll(_pcb, &_tcp_poll, 1);
606 }
607 return *this;
608}

614AsyncClient & AsyncClient::operator+=(const AsyncClient &other) {
615 if(next == NULL){
616 next = (AsyncClient*)(&other);
617 next->prev = this;
618 } else {
619 AsyncClient *c = next;
620 while(c->next != NULL) {
621 c = c->next;
622 }
623 c->next =(AsyncClient*)(&other);
624 c->next->prev = c;
625 }
626 return *this;
627}

821void AsyncClient::_allocate_closed_slot(){
822 xSemaphoreTake(_slots_lock, portMAX_DELAY);
823 uint32_t closed_slot_min_index = 0;
824 for (int i = 0; i < _number_of_closed_slots; ++ i) {
825 if ((_closed_slot == -1 || _closed_slots[i] <= closed_slot_min_index) && _closed_slots[i] != 0) {
826 closed_slot_min_index = _closed_slots[i];
827 _closed_slot = i;
828 }
829 }
830 if (_closed_slot != -1) {
831 _closed_slots[_closed_slot] = 0;
832 }
833 xSemaphoreGive(_slots_lock);
834}

922int8_t AsyncClient::_recv(tcp_pcb* pcb, pbuf* pb, int8_t err) {
923 while(pb != NULL) {
924 _rx_last_packet = millis();
925 //we should not ack before we assimilate the data
926 _ack_pcb = true;
927 pbuf *b = pb;
928 pb = b->next;
929 b->next = NULL;
930 if(_pb_cb){
931 _pb_cb(_pb_cb_arg, this, b);
932 } else {
933 if(_recv_cb) {
934 _recv_cb(_recv_cb_arg, this, b->payload, b->len);
935 }
936 if(!_ack_pcb) {
937 _rx_ack_len += b->len;
938 } else if(_pcb) {
939 _tcp_recved(_pcb, _closed_slot, b->len);
940 }
941 pbuf_free(b);
942 }
943 }

947int8_t AsyncClient::_poll(tcp_pcb* pcb){
948 if(!_pcb){
949 log_w("pcb is NULL");
950 return ERR_OK;
951 }
952 if(pcb != _pcb){
953 log_e("0x%08x != 0x%08x", (uint32_t)pcb, (uint32_t)_pcb);
954 return ERR_OK;
955 }
956
957 uint32_t now = millis();
958
959 // ACK Timeout
960 if(_pcb_busy && _ack_timeout && (now - _pcb_sent_at) >= _ack_timeout){
961 _pcb_busy = false;
962 log_w("ack timeout %d", pcb->state);
963 if(_timeout_cb)
964 _timeout_cb(_timeout_cb_arg, this, (now - _pcb_sent_at));
965 return ERR_OK;
966 }
967 // RX Timeout
968 if(_rx_since_timeout && (now - _rx_last_packet) >= (_rx_since_timeout * 1000)){
969 log_w("rx timeout %d", pcb->state);
970 _close();
971 return ERR_OK;
972 }
973 // Everything is fine
974 if(_poll_cb) {
975 _poll_cb(_poll_cb_arg, this);
976 }
977 return ERR_OK;
