58void calibrate(imu_structs_t imu_objs)
59{
60 uint8_t reg = 0;
61 while (calib_done == 0)
62 {
63 HAL_I2C_Mem_Read(imu_objs.hi2c, imu_objs.dev_addr, CALIB_REG, 1, &reg, 1, 1000);
64 HAL_Delay(50);
65 reg = reg & 3; // Bottom 2 bits correspond to mag calibration
66 if (reg == 3)
67 {
68 calib_done = 1;
69 HAL_GPIO_WritePin(GPIOC, led_Pin, GPIO_PIN_RESET);
70 HAL_Delay(500);
71 HAL_GPIO_WritePin(GPIOC, led_Pin, GPIO_PIN_SET);
72 }
73 }
74}

82int16_t temperature(imu_structs_t imu_objs)
83{
84 uint8_t buf[12] = {0};
85 int16_t val;
86 int16_t temp_c;
87
88 ret = HAL_I2C_Mem_Read(imu_objs.hi2c, imu_objs.dev_addr, TEMP_REG, 1, buf, 1, 1000);
89 if (ret != HAL_OK) // If HAL transmit failed, show error message
90 {
91 strcpy((char*)buf, "Error RX \r\n");
92 }
93 else
94 {
95 val = (int16_t)buf[0];
96 if (val > 0x7F) // Check if num > than max positive signed number
97 {
98 val |= 0xF00; // If yes, sign extend a nibble of 1's in front of num
99 }
100 temp_c = val;
101 sprintf((char*)buf,"%d C \r\n", temp_c);
102 }
103 //CDC_Transmit_FS((uint8_t*)buf,strlen((char*)buf));
104 return temp_c;
105}
